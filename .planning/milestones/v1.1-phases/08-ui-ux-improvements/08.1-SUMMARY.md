import { classifyAlert, AlertSeverity } from '@/lib/severity';
import { AlertGroupDisplay, groupAlerts } from '@/components/live/AlertGroup';

// Classify an alert
const severity = classifyAlert('person', 85, 'front_steps');

// Group multiple alerts

  <AlertGroupDisplay
