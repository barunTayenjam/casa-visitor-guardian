### 4. Auto-Optimization Toggle
- [ ] Add auto-optimization setting
- [ ] Implement automatic quality adjustment
- [ ] Show optimization status in UI
- [ ] Allow manual override

**Auto-Optimization Logic:**
```typescript
async function autoOptimize() {
  const usage = await getResourceUsage();
  
  if (usage.cpu > 80 || usage.memory > 1600) {
    console.log('High resource usage detected, optimizing...');
    
    // Increase detection interval
    await updateConfig('detection.interval', 5000);
    
    // Reduce stream FPS
    await updateCameraPreset('all', 'low');
    
    // Increase event cleanup frequency
    await updateConfig('retention.maxEvents', 300);
    
    return { action: 'reduced', reason: 'high_load' };
  }
  
  if (usage.cpu < 40 && usage.memory < 1000) {
    console.log('Resources available, optimizing for quality...');
    
    // Restore normal settings
    await updateConfig('detection.interval', 2000);
    await updateCameraPreset('all', 'medium');
    await updateConfig('retention.maxEvents', 1000);
    
    return { action: 'restored', reason: 'low_load' };
  }
  
  return { action: 'none', reason: 'stable' };
}
```
