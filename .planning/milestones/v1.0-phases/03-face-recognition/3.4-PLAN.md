```typescript
import { AppDataSource } from '../database.js';
import { UnknownFaceDetection, UnknownFaceAlert } from '../models/UnknownFace.js';
import { Event } from '../models/Event.js';

export class UnknownFaceService {
  private unknownFaceRepo = AppDataSource.getRepository(UnknownFaceDetection);
  private alertRepo = AppDataSource.getRepository(UnknownFaceAlert);
  private eventRepo = AppDataSource.getRepository(Event);

  async trackUnknownFace(
    eventId: string,
    faceId: string,
    recognitionResult: {
      name: string;
      confidence: number;
      similarity?: number;
      threshold?: number;
      matchedVisitorId?: string;
      matchedVisitorName?: string;
    },
    faceData: {
      bbox: any;
      confidence: number;
      embedding?: number[];
    }
  ): Promise<UnknownFaceDetection> {
    // Get event details
    const event = await this.eventRepo.findOne({ where: { id: eventId } });
    if (!event) {
      throw new Error('Event not found');
    }

    // Check if already tracked
    const existing = await this.unknownFaceRepo.findOne({
      where: { eventId, faceId }
    });

    if (existing) {
      // Update existing record
      existing.detectionCount++;
      existing.lastSeen = new Date();
      if (recognitionResult.similarity !== undefined) {
        existing.similarityScore = recognitionResult.similarity;
      }
      if (recognitionResult.threshold !== undefined) {
        existing.similarityThreshold = recognitionResult.threshold;
      }
      if (recognitionResult.matchedVisitorId) {
        existing.matchedVisitorId = recognitionResult.matchedVisitorId;
      }
      if (recognitionResult.matchedVisitorName) {
        existing.matchedVisitorName = recognitionResult.matchedVisitorName;
      }

      await this.unknownFaceRepo.save(existing);
      return existing;
    }

    // Create new unknown face detection
    const detection = this.unknownFaceRepo.create({
      eventId,
      faceId,
      bbox: faceData.bbox,
      confidence: faceData.confidence,
      embeddingVector: faceData.embedding || null,
      similarityScore: recognitionResult.similarity || null,
      similarityThreshold: recognitionResult.threshold || null,
      matchedVisitorId: recognitionResult.matchedVisitorId || null,
      matchedVisitorName: recognitionResult.matchedVisitorName || null,
      cameraId: event.cameraId,
      timestamp: event.timestamp,
      imagePath: event.imagePath,
      status: recognitionResult.name === 'unknown' ? 'unknown' : 'identified',
      firstSeen: event.timestamp,
      lastSeen: event.timestamp
    });

    await this.unknownFaceRepo.save(detection);

    // Create alert if truly unknown
    if (recognitionResult.name === 'unknown') {
      await this.createAlert(detection);
    }

    // Update pattern tracking
    await this.updatePatternTracking(detection);

    return detection;
  }

  async createAlert(detection: UnknownFaceDetection): Promise<UnknownFaceAlert> {
    // Determine severity based on time and confidence
    const hour = new Date(detection.timestamp).getHours();
    const isNightTime = hour >= 22 || hour <= 6;
    const isHighConfidence = detection.confidence > 80;

    let severity: 'low' | 'medium' | 'high' | 'critical' = 'medium';
    let alertType = 'unknown_face';
    let message = 'Unknown face detected';

    if (isNightTime && isHighConfidence) {
      severity = 'critical';
      alertType = 'night_time';
      message = 'Unknown face detected at night (high confidence)';
    } else if (isNightTime) {
      severity = 'high';
      alertType = 'night_time';
      message = 'Unknown face detected at night';
    } else if (isHighConfidence) {
      severity = 'medium';
      message = 'Unknown face detected (high confidence)';
    }

    // Check if this is a frequent unknown face
    const recentDetections = await this.unknownFaceRepo.count({
      where: {
        cameraId: detection.cameraId,
        status: 'unknown'
      }
    });

    if (recentDetections > 5) {
      severity = 'high';
      alertType = 'frequent_unknown';
      message = `Frequent unknown face detected (${recentDetections} times)`;
    }

    const alert = this.alertRepo.create({
      detectionId: detection.id,
      alertType,
      severity,
      message,
      status: 'active'
    });

    await this.alertRepo.save(alert);

    // Send notification (via notification service)
    await this.sendNotification(alert, detection);

    return alert;
  }

  async markAsKnown(
    detectionId: string,
    visitorId: string,
    visitorName: string,
    userId: string
  ): Promise<void> {
    const detection = await this.unknownFaceRepo.findOne({
      where: { id: detectionId }
    });

    if (!detection) {
      throw new Error('Unknown face detection not found');
    }

    detection.status = 'identified';
    detection.markedAsVisitorId = visitorId;
    detection.markedAsVisitorName = visitorName;
    detection.markedAt = new Date();
    detection.markedBy = userId;

    await this.unknownFaceRepo.save(detection);

    // Dismiss associated alerts
    await this.alertRepo.update(
      { detectionId, status: 'active' },
      { status: 'dismissed' }
    );
  }

  async dismissDetection(detectionId: string, userId: string): Promise<void> {
    const detection = await this.unknownFaceRepo.findOne({
      where: { id: detectionId }
    });

    if (!detection) {
      throw new Error('Unknown face detection not found');
    }

    detection.status = 'dismissed';
    await this.unknownFaceRepo.save(detection);

    // Dismiss associated alerts
    await this.alertRepo.update(
      { detectionId, status: 'active' },
      { status: 'dismissed' }
    );
  }

  async acknowledgeAlert(alertId: string, userId: string): Promise<void> {
    await this.alertRepo.update(
      { id: alertId },
      {
        status: 'acknowledged',
        acknowledgedAt: new Date(),
        acknowledgedBy: userId
      }
    );
  }

  async getUnknownFaces(params: {
    status?: string;
    cameraId?: string;
    startDate?: Date;
    endDate?: Date;
    page?: number;
    limit?: number;
  }): Promise<{ detections: UnknownFaceDetection[]; total: number }> {
    const queryBuilder = this.unknownFaceRepo.createQueryBuilder('d');

    if (params.status) {
      queryBuilder.andWhere('d.status = :status', { status: params.status });
    }

    if (params.cameraId) {
      queryBuilder.andWhere('d.cameraId = :cameraId', { cameraId: params.cameraId });
    }

    if (params.startDate) {
      queryBuilder.andWhere('d.timestamp >= :startDate', { startDate: params.startDate });
    }

    if (params.endDate) {
      queryBuilder.andWhere('d.timestamp <= :endDate', { endDate: params.endDate });
    }

    const page = params.page || 1;
    const limit = params.limit || 20;

    queryBuilder
      .orderBy('d.timestamp', 'DESC')
      .skip((page - 1) * limit)
      .take(limit);

    const [detections, total] = await queryBuilder.getManyAndCount();

    return { detections, total };
  }

  async getActiveAlerts(): Promise<UnknownFaceAlert[]> {
    return this.alertRepo.find({
      where: { status: 'active' },
      relations: ['detection'],
      order: { createdAt: 'DESC' }
    });
  }

  private async updatePatternTracking(detection: UnknownFaceDetection): Promise<void> {
    if (!detection.embeddingVector) {
      return;
    }

    // Generate pattern hash (simplified - could use better clustering)
    const hash = this.generatePatternHash(detection.embeddingVector);

    // This would update the unknown_face_patterns table
    // Implementation depends on clustering algorithm
  }

  private generatePatternHash(embedding: number[]): string {
    // Simplified hash generation
    // In production, use proper clustering (e.g., DBSCAN, K-means)
    const buckets = embedding.map(v => Math.floor(v * 10));
    return buckets.join('-');
  }

  private async sendNotification(alert: UnknownFaceAlert, detection: UnknownFaceDetection): Promise<void> {
    // Integrate with notification service from Phase 2
    // For now, just mark as sent
    alert.notificationSent = true;
    alert.notificationSentAt = new Date();
    alert.notificationMethod = ['push'];

    await this.alertRepo.save(alert);
  }
}

export const unknownFaceService = new UnknownFaceService();
```

```typescript
  const markAsKnownMutation = useMutation({
    mutationFn: ({ id, visitorId, visitorName }: any) =>
      apiService.markUnknownAsKnown(id, visitorId, visitorName),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['unknown-faces'] });
      setIsMarkDialogOpen(false);
    },
  });

  const dismissMutation = useMutation({
    mutationFn: (id: string) => apiService.dismissUnknownFace(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['unknown-faces'] });
    },
  });

  if (isLoading) return <div>Loading unknown faces...</div>;

  return (
    <div className="p-8 max-w-7xl mx-auto">
      <div className="mb-8">
        <h1 className="text-3xl font-bold">Unknown Faces</h1>
        <p className="text-muted-foreground mt-2">
          Review and manage unknown face detections
        </p>
      </div>

      {/* Alert Stats */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-8">
        <Alert>
          <AlertTriangle className="h-4 w-4" />
          <AlertDescription>
            {unknownFaces?.length || 0} unknown faces detected
          </AlertDescription>
        </Alert>
      </div>

      {/* Unknown Faces Grid */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
        {unknownFaces?.map((face: UnknownFace) => (
          <UnknownFaceCard
            key={face.id}
            face={face}
            onMarkKnown={() => {
              setSelectedFace(face);
              setIsMarkDialogOpen(true);
            }}
            onDismiss={() => dismissMutation.mutate(face.id)}
          />
        ))}
      </div>

      {/* Mark as Known Dialog */}
      <Dialog open={isMarkDialogOpen} onOpenChange={setIsMarkDialogOpen}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Mark as Known Visitor</DialogTitle>
          </DialogHeader>
          {selectedFace && (
            <MarkAsKnownForm
              face={selectedFace}
              onSubmit={(visitorId, visitorName) => {
                markAsKnownMutation.mutate({
                  id: selectedFace.id,
                  visitorId,
                  visitorName
                });
              }}
            />
          )}
        </DialogContent>
      </Dialog>
    </div>
  );
};

const UnknownFaceCard: React.FC<{
```
