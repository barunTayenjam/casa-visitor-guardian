## Objective

Implement comprehensive database cleanup operations including foreign key cascade deletes, orphaned record removal, scheduled vacuum operations, and index optimization to maintain database performance and integrity.

## Success Criteria

1. Foreign key relationships properly cascade deletes
2. Orphaned records detected and cleaned up
3. Database vacuum runs weekly (Sunday 4 AM)
4. Index optimization runs monthly
5. Database integrity checks pass
6. Storage reclaimed after hard deletes
7. Query performance maintained at scale
8. Cleanup logged to audit trail

## Current State Analysis

**Database Schema:**
- 17 tables with various foreign key relationships
- `detection_files` table (0 rows) relates to `events` table
- `events` table (1,050+ rows) has no foreign key to `detection_files`
- No cascade delete rules defined
- No scheduled vacuum operations

**Issues:**
- Orphaned records possible when files deleted
- No database maintenance schedule
- No storage reclamation after deletes
- Bloat accumulation from updates/deletes
- Index statistics may be stale

### Step 1: Database Schema Foreign Key Enhancement

**File:** `database/migrations/012_add_foreign_key_cascades.sql`

```sql
-- Add foreign key constraints with CASCADE deletes

-- events table should CASCADE when detection_files deleted
-- (This ensures event records are cleaned up when files are hard deleted)

-- First, add a detection_file_uuid column to events if not exists
DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM information_schema.columns
        WHERE table_name = 'events' AND column_name = 'detection_file_uuid'
    ) THEN
        ALTER TABLE events ADD COLUMN detection_file_uuid UUID;
        CREATE INDEX idx_events_detection_file_uuid ON events(detection_file_uuid);
    END IF;
END $$;

-- Add foreign key with CASCADE
DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM information_schema.table_constraints
        WHERE constraint_name = 'fk_events_detection_file'
    ) THEN
        ALTER TABLE events
        ADD CONSTRAINT fk_events_detection_file
        FOREIGN KEY (detection_file_uuid)
        REFERENCES detection_files(file_uuid)
        ON DELETE CASCADE;
    END IF;
END $$;

-- visitor_timeline cleanup - remove orphaned visitor records
-- that have no corresponding events

-- Add a cleanup function for orphaned records
CREATE OR REPLACE FUNCTION cleanup_orphaned_records()
RETURNS TABLE (
    table_name TEXT,
    records_deleted BIGINT
) AS $$
BEGIN
    -- Clean up orphaned visitor_timeline entries
    -- (visitors with no events in the last 90 days)
    RETURN QUERY
    WITH orphaned_visitors AS (
        SELECT vt.id
        FROM visitor_timeline vt
        LEFT JOIN events e ON e.visitor_id = vt.id
        WHERE e.id IS NULL
        AND vt.created_at < CURRENT_TIMESTAMP - INTERVAL '90 days'
    )
    DELETE FROM visitor_timeline
    WHERE id IN (SELECT id FROM orphaned_visitors)
    RETURNING 'visitor_timeline'::TEXT, COUNT(*)::BIGINT;

    -- Clean up orphaned batch_jobs (older than 30 days, status = completed)
    RETURN QUERY
    DELETE FROM batch_jobs
    WHERE status = 'completed'
    AND completed_at IS NOT NULL
    AND completed_at < CURRENT_TIMESTAMP - INTERVAL '30 days'
    RETURNING 'batch_jobs'::TEXT, COUNT(*)::BIGINT;

    -- Clean up old user_sessions (older than 7 days)
    RETURN QUERY
    DELETE FROM user_sessions
    WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '7 days'
    RETURNING 'user_sessions'::TEXT, COUNT(*)::BIGINT;

    -- Clean up old audit_logs (older than 90 days, keep critical ones)
    RETURN QUERY
    DELETE FROM audit_logs
    WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '90 days'
    AND action NOT IN ('retention_cleanup', 'retention_policy_update', 'security_event')
    RETURNING 'audit_logs'::TEXT, COUNT(*)::BIGINT;

    -- Clean up old notification_logs (older than 30 days)
    RETURN QUERY
    DELETE FROM notification_logs
    WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '30 days'
    RETURNING 'notification_logs'::TEXT, COUNT(*)::BIGINT;

END;
$$ LANGUAGE plpgsql;

-- Create function for database vacuum and analyze
CREATE OR REPLACE FUNCTION vacuum_and_analyze_tables()
RETURNS VOID AS $$
BEGIN
    -- Vacuum detection_files (reclaims space after hard deletes)
    RAISE NOTICE 'Vacuuming detection_files...';
    VACUUM ANALYZE detection_files;

    -- Vacuum events (large table, needs regular maintenance)
    RAISE NOTICE 'Vacuuming events...';
    VACUUM ANALYZE events;

    -- Vacuum visitor_timeline
    RAISE NOTICE 'Vacuuming visitor_timeline...';
    VACUUM ANALYZE visitor_timeline;

    -- Vacuum batch_jobs
    RAISE NOTICE 'Vacuuming batch_jobs...';
    VACUUM ANALYZE batch_jobs;

    -- Analyze all tables to update statistics
    RAISE NOTICE 'Analyzing all tables...';
    ANALYZE;

    RAISE NOTICE 'Vacuum and analyze completed';
END;
$$ LANGUAGE plpgsql;

-- Create function for index optimization
CREATE OR REPLACE FUNCTION optimize_indexes()
RETURNS TABLE (
    index_name TEXT,
    index_size_before TEXT,
    index_size_after TEXT
) AS $$
DECLARE
    idx RECORD;
    size_before TEXT;
    size_after TEXT;
BEGIN
    -- Reindex all indexes on detection_files
    FOR idx IN
        SELECT indexname
        FROM pg_indexes
        WHERE tablename = 'detection_files'
    LOOP
        -- Get size before
        SELECT pg_size_pretty(pg_relation_size(indexrelid::regclass))
        INTO size_before
        FROM pg_stat_user_indexes
        WHERE schemaname = 'public' AND indexname = idx.indexname;

        -- Reindex
        EXECUTE 'REINDEX INDEX CONCURRENTLY ' || idx.indexname;

        -- Get size after
        SELECT pg_size_pretty(pg_relation_size(indexrelid::regclass))
        INTO size_after
        FROM pg_stat_user_indexes
        WHERE schemaname = 'public' AND indexname = idx.indexname;

        RETURN NEXT;
    END LOOP;

    -- Reindex indexes on events table
    FOR idx IN
        SELECT indexname
        FROM pg_indexes
        WHERE tablename = 'events'
    LOOP
        SELECT pg_size_pretty(pg_relation_size(indexrelid::regclass))
        INTO size_before
        FROM pg_stat_user_indexes
        WHERE schemaname = 'public' AND indexname = idx.indexname;

        EXECUTE 'REINDEX INDEX CONCURRENTLY ' || idx.indexname;

        SELECT pg_size_pretty(pg_relation_size(indexrelid::regclass))
        INTO size_after
        FROM pg_stat_user_indexes
        WHERE schemaname = 'public' AND indexname = idx.indexname;

        RETURN NEXT;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

-- Create function to check database integrity
CREATE OR REPLACE FUNCTION check_database_integrity()
RETURNS TABLE (
    table_name TEXT,
    issue_count BIGINT,
    issue_type TEXT
) AS $$
BEGIN
    -- Check for orphaned detection_files (no corresponding event)
    RETURN QUERY
    SELECT 'detection_files'::TEXT, COUNT(*)::BIGINT, 'orphaned_files'::TEXT
    FROM detection_files df
    LEFT JOIN events e ON e.detection_file_uuid = df.file_uuid
    WHERE e.id IS NULL
    AND df.created_at < CURRENT_TIMESTAMP - INTERVAL '1 day'
    HAVING COUNT(*) > 0;

    -- Check for events with missing detection files
    RETURN QUERY
    SELECT 'events'::TEXT, COUNT(*)::BIGINT, 'missing_files'::TEXT
    FROM events e
    LEFT JOIN detection_files df ON df.file_uuid = e.detection_file_uuid
    WHERE e.detection_file_uuid IS NOT NULL
    AND df.file_uuid IS NULL
    HAVING COUNT(*) > 0;

    -- Check for orphaned user_sessions (no valid user)
    RETURN QUERY
    SELECT 'user_sessions'::TEXT, COUNT(*)::BIGINT, 'orphaned_sessions'::TEXT
    FROM user_sessions us
    LEFT JOIN users u ON u.id = us.user_id
    WHERE u.id IS NULL
    HAVING COUNT(*) > 0;

END;
$$ LANGUAGE plpgsql;

-- Comments for documentation
COMMENT ON FUNCTION cleanup_orphaned_records() IS 'Removes orphaned records from multiple tables';
COMMENT ON FUNCTION vacuum_and_analyze_tables() IS 'Vacuums and analyzes tables to reclaim space and update statistics';
COMMENT ON FUNCTION optimize_indexes() IS 'Rebuilds indexes to optimize performance';
COMMENT ON FUNCTION check_database_integrity() IS 'Checks for orphaned records and integrity issues';
```

### Step 2: Database Cleanup Service

**File:** `server/src/services/databaseCleanupService.ts`

```typescript
import { AppDataSource } from '../database.js';
import cron from 'node-cron';

export interface DatabaseCleanupResult {
  success: boolean;
  vacuumCompleted: boolean;
  orphanedRecordsCleaned: number;
  indexesOptimized: number;
  integrityCheckPassed: boolean;
  issues: string[];
  duration: number; // milliseconds
}

export class DatabaseCleanupService {
  /**
   * Run full database cleanup
   */
  async runFullCleanup(): Promise<DatabaseCleanupResult> {
    const startTime = Date.now();
    const result: DatabaseCleanupResult = {
      success: false,
      vacuumCompleted: false,
      orphanedRecordsCleaned: 0,
      indexesOptimized: 0,
      integrityCheckPassed: false,
      issues: [],
      duration: 0
    };

    try {
      console.log('[DatabaseCleanup] Starting full database cleanup...');

      // Step 1: Clean orphaned records
      const orphanedResult = await this.cleanupOrphanedRecords();
      result.orphanedRecordsCleaned = orphanedResult.totalDeleted;

      // Step 2: Vacuum and analyze tables
      await this.vacuumAndAnalyze();
      result.vacuumCompleted = true;

      // Step 3: Check integrity
      const integrityResult = await this.checkIntegrity();
      result.integrityCheckPassed = integrityResult.issues.length === 0;

      if (!result.integrityCheckPassed) {
        result.issues.push(...integrityResult.issues);
      }

      result.success = true;
      result.duration = Date.now() - startTime;

      console.log(`[DatabaseCleanup] Completed in ${result.duration}ms`, result);

      // Log to audit
      await this.logCleanupToAudit(result);

      return result;
    } catch (error) {
      console.error('[DatabaseCleanup] Error:', error);
      result.success = false;
      result.issues.push(error instanceof Error ? error.message : String(error));
      result.duration = Date.now() - startTime;
      return result;
    }
  }

  /**
   * Clean orphaned records
   */
  private async cleanupOrphanedRecords(): Promise<{ totalDeleted: number }> {
    try {
      const result = await AppDataSource.query(
        'SELECT * FROM cleanup_orphaned_records()'
      );

      const totalDeleted = result.reduce((sum: number, r: any) => sum + (parseInt(r.records_deleted) || 0), 0);

      console.log(`[DatabaseCleanup] Cleaned ${totalDeleted} orphaned records`);

      return { totalDeleted };
    } catch (error) {
      console.error('[DatabaseCleanup] Error cleaning orphaned records:', error);
      throw error;
    }
  }

  /**
   * Vacuum and analyze tables
   */
  private async vacuumAndAnalyze(): Promise<void> {
    try {
      console.log('[DatabaseCleanup] Starting vacuum and analyze...');
      await AppDataSource.query('SELECT vacuum_and_analyze_tables()');
      console.log('[DatabaseCleanup] Vacuum and analyze completed');
    } catch (error) {
      console.error('[DatabaseCleanup] Vacuum error:', error);
      throw error;
    }
  }

  /**
   * Optimize indexes
   */
  async optimizeIndexes(): Promise<{ indexesOptimized: number }> {
    try {
      console.log('[DatabaseCleanup] Starting index optimization...');
      const result = await AppDataSource.query('SELECT * FROM optimize_indexes()');
      const indexesOptimized = result.length;
      console.log(`[DatabaseCleanup] Optimized ${indexesOptimized} indexes`);
      return { indexesOptimized };
    } catch (error) {
      console.error('[DatabaseCleanup] Index optimization error:', error);
      throw error;
    }
  }

  /**
   * Check database integrity
   */
  private async checkIntegrity(): Promise<{ issues: string[] }> {
    try {
      const result = await AppDataSource.query('SELECT * FROM check_database_integrity()');

      const issues: string[] = [];
      for (const row of result) {
        if (parseInt(row.issue_count) > 0) {
          issues.push(`${row.table_name}: ${row.issue_count} ${row.issue_type}`);
        }
      }

      if (issues.length > 0) {
        console.warn('[DatabaseCleanup] Integrity issues found:', issues);
      } else {
        console.log('[DatabaseCleanup] Integrity check passed');
      }

      return { issues };
    } catch (error) {
      console.error('[DatabaseCleanup] Integrity check error:', error);
      throw error;
    }
  }

  /**
   * Get database statistics
   */
  async getDatabaseStats(): Promise<{
    tableSize: string;
    indexSize: string;
    totalSize: string;
    bloatRatio: number;
  }> {
    try {
      const stats = await AppDataSource.query(`
        SELECT
          pg_size_pretty(pg_total_relation_size('detection_files')) as table_size,
          pg_size_pretty(pg_indexes_size('detection_files')) as index_size,
          pg_size_pretty(pg_total_relation_size('detection_files') + pg_indexes_size('detection_files')) as total_size
      `);

      // Calculate bloat ratio (this is a simplified check)
      const bloatResult = await AppDataSource.query(`
        SELECT
          COUNT(*) as total_rows,
          COUNT(*) FILTER (WHERE is_deleted = true) as deleted_rows
        FROM detection_files
      `);

      const totalRows = parseInt(bloatResult[0].total_rows) || 1;
      const deletedRows = parseInt(bloatResult[0].deleted_rows) || 0;
      const bloatRatio = (deletedRows / totalRows) * 100;

      return {
        tableSize: stats[0].table_size,
        indexSize: stats[0].index_size,
        totalSize: stats[0].total_size,
        bloatRatio
      };
    } catch (error) {
      console.error('[DatabaseCleanup] Error getting database stats:', error);
      throw error;
    }
  }

  /**
   * Log cleanup operation to audit
   */
  private async logCleanupToAudit(result: DatabaseCleanupResult): Promise<void> {
    try {
      await AppDataSource.query(
        `INSERT INTO audit_logs (user_id, action, details, ip_address)
         VALUES ($1, $2, $3, $4)`,
        [
          null, // System operation
          'database_cleanup',
          JSON.stringify(result),
          'system'
        ]
      );
    } catch (error) {
      console.error('[DatabaseCleanup] Audit log error:', error);
    }
  }

  /**
   * Get cleanup status
   */
  async getCleanupStatus(): Promise<{
    lastCleanup: DatabaseCleanupResult | null;
    nextVacuum: Date;
    nextIndexOptimization: Date;
  }> {
    // Get last cleanup from audit logs
    const lastCleanupResult = await AppDataSource.query(
      `SELECT details FROM audit_logs
       WHERE action = 'database_cleanup'
       ORDER BY created_at DESC
       LIMIT 1`
    );

    const lastCleanup = lastCleanupResult.length > 0
      ? JSON.parse(lastCleanupResult[0].details) as DatabaseCleanupResult
      : null;

    // Next vacuum is Sunday 4 AM
    const nextVacuum = this.getNextSunday4AM();

    // Next index optimization is 1st of next month
    const nextIndexOptimization = new Date();
    nextIndexOptimization.setMonth(nextIndexOptimization.getMonth() + 1);
    nextIndexOptimization.setDate(1);
    nextIndexOptimization.setHours(4, 0, 0, 0);

    return {
      lastCleanup,
      nextVacuum,
      nextIndexOptimization
    };
  }

  /**
   * Get next Sunday 4 AM
   */
  private getNextSunday4AM(): Date {
    const now = new Date();
    const dayOfWeek = now.getDay();
    const daysUntilSunday = (7 - dayOfWeek) % 7 || 7; // If today is Sunday, next week

    const nextSunday = new Date(now);
    nextSunday.setDate(now.getDate() + daysUntilSunday);
    nextSunday.setHours(4, 0, 0, 0);

    return nextSunday;
  }
}

export const databaseCleanupService = new DatabaseCleanupService();
```

```typescript
import express from 'express';
import { databaseCleanupService } from '../services/databaseCleanupService.js';

const router = express.Router();

// POST /api/database/cleanup - Trigger manual cleanup
router.post('/cleanup', async (req, res) => {
  try {
    // Check admin permissions
    if (req.user?.role !== 'admin') {
      return res.status(403).json({ error: 'Admin access required' });
    }

    console.log('[API] Manual database cleanup triggered by', req.user?.id);

    const result = await databaseCleanupService.runFullCleanup();

    res.json({
      message: 'Database cleanup completed',
      result
    });
  } catch (error) {
    console.error('[API] Manual cleanup error:', error);
    res.status(500).json({ error: 'Cleanup failed' });
  }
});

// POST /api/database/optimize-indexes - Trigger index optimization
router.post('/optimize-indexes', async (req, res) => {
  try {
    // Check admin permissions

      message: 'Index optimization completed',

  } catch (error) {
    console.error('[API] Index optimization error:', error);
```
