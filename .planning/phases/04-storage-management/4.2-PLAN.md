## Objective

Build an automated cleanup service that enforces retention policies by deleting old files and database records, with proper soft-delete cascade, archiving, and integrity validation.

## Success Criteria

1. Cleanup service runs daily at 3 AM via cron
2. Files are deleted according to retention policies
3. Soft delete precedes hard delete (7-day grace period)
4. Archive workflow (move files to archive/ directory)
5. Database records cleaned up when files deleted
6. Integrity validation using SHA-256 hashes
7. Cleanup operations logged to audit trail
8. Manual cleanup trigger available via API
9. Progress tracking for long-running cleanup jobs

import { Entity, PrimaryColumn, Column, Index, CreateDateColumn } from 'typeorm';

@Entity('detection_files')
@Index(['fileType'])
@Index(['cameraId'])
@Index(['captureTimestamp'])
@Index(['createdAt'])
@Index(['isArchived'])
@Index(['isDeleted'])
export class DetectionFile {
  @PrimaryColumn('uuid')
  fileUuid!: string;

  @Column({ type: 'text' })
  fileType!: string; // 'event_face', 'event_motion', 'snapshot', 'batch_result', 'temp'

  @Column({ type: 'text' })
  cameraId!: string; // 'cam1', 'cam2', etc.

  @Column({ type: 'text' })
  storagePath!: string; // Absolute path to file

  @Column({ type: 'bigint' })
  fileSize!: number; // File size in bytes

  @Column({ type: 'text' })
  fileHash!: string; // SHA-256 hash for integrity

  @Column({ type: 'timestamp with time zone', nullable: true })
  captureTimestamp?: Date; // When the image was captured

  @Column({ type: 'jsonb', nullable: true })
  metadata?: {
    // Detection metadata (objects, faces, confidence, etc.)
    objects?: Array<{ label: string; confidence: number }>;
    faces?: number;
    persons?: number;
    confidence?: number;
  };

  // Soft delete flags
  @Column({ type: 'boolean', default: false })
  isDeleted!: boolean;

  @Column({ type: 'timestamp with time zone', nullable: true })
  deletedAt?: Date;

  @Column({ type: 'boolean', default: false })
  isArchived!: boolean;

  @Column({ type: 'timestamp with time zone', nullable: true })
  archivedAt?: Date;

  @CreateDateColumn({ name: 'created_at' })
  createdAt!: Date;
}

### Step 2: Retention Cleanup Service

**File:** `server/src/services/retentionCleanupService.ts`

```typescript
import { AppDataSource } from '../database.js';
import { DetectionFile } from '../models/DetectionFile.js';
import { RetentionPolicy } from '../models/RetentionPolicy.js';
import { retentionPolicyService } from './retentionPolicyService.js';
import * as fs from 'fs/promises';
import * as path from 'path';
import * as crypto from 'crypto';
import { createHash } from 'crypto';

const detectionFileRepository = AppDataSource.getRepository(DetectionFile);

export interface CleanupResult {
  success: boolean;
  processed: number;
  softDeleted: number;
  archived: number;
  hardDeleted: number;
  errors: number;
  errorMessage?: string;
  duration: number; // milliseconds
}

export class RetentionCleanupService {
  /**
   * Main cleanup method - enforces all retention policies
   */
  async enforcePolicies(): Promise<CleanupResult> {
    const startTime = Date.now();
    const result: CleanupResult = {
      success: false,
      processed: 0,
      softDeleted: 0,
      archived: 0,
      hardDeleted: 0,
      errors: 0,
      duration: 0
    };

    try {
      console.log('[RetentionCleanup] Starting policy enforcement...');

      // Get all active policies
      const policies = await AppDataSource
        .getRepository(RetentionPolicy)
        .find({ where: { isActive: true } });

      console.log(`[RetentionCleanup] Found ${policies.length} active policies`);

      // Process each policy
      for (const policy of policies) {
        const policyResult = await this.enforceSinglePolicy(policy);

        result.processed += policyResult.processed;
        result.softDeleted += policyResult.softDeleted;
        result.archived += policyResult.archived;
        result.hardDeleted += policyResult.hardDeleted;
        result.errors += policyResult.errors;
      }

      // Vacuum database to reclaim space
      await this.vacuumDatabase();

      result.success = true;
      result.duration = Date.now() - startTime;

      console.log(`[RetentionCleanup] Completed in ${result.duration}ms`, result);

      // Log to audit
      await this.logCleanupToAudit(result);

      return result;
    } catch (error) {
      console.error('[RetentionCleanup] Error:', error);
      result.success = false;
      result.errorMessage = error instanceof Error ? error.message : String(error);
      result.duration = Date.now() - startTime;
      return result;
    }
  }

  /**
   * Enforce a single retention policy
   */
  async enforceSinglePolicy(policy: RetentionPolicy): Promise<CleanupResult> {
    const result: CleanupResult = {
      success: false,
      processed: 0,
      softDeleted: 0,
      archived: 0,
      hardDeleted: 0,
      errors: 0,
      duration: 0
    };

    try {
      const cameraId = policy.camera === 'global' ? 'all' : policy.camera;

      // Step 1: Soft delete files older than retention period
      await this.softDeleteOldFiles(policy, result);

      // Step 2: Archive soft-deleted files past grace period
      await this.archiveSoftDeletedFiles(policy, result);

      // Step 3: Hard delete archived files past archive retention
      await this.hardDeleteArchivedFiles(policy, result);

      result.success = true;
      return result;
    } catch (error) {
      console.error(`[RetentionCleanup] Error enforcing policy for ${cameraId}:`, error);
      result.errors++;
      result.errorMessage = error instanceof Error ? error.message : String(error);
      return result;
    }
  }

  /**
   * Soft delete files older than retention period
   */
  private async softDeleteOldFiles(policy: RetentionPolicy, result: CleanupResult): Promise<void> {
    const cameraId = policy.camera;

    // Get file types and their retention periods
    const fileTypes = [
      { type: 'event_face', days: policy.detectionsDays },
      { type: 'event_motion', days: policy.detectionsDays },
      { type: 'event_alert', days: policy.alertsDays },
      { type: 'snapshot', days: policy.snapshotsDays },
      { type: 'preview', days: policy.previewsDays },
      { type: 'batch_result', days: policy.batchResultsDays },
      { type: 'temp', days: policy.tempFilesDays }
    ];

    for (const { type, days } of fileTypes) {
      if (days === null || days === undefined || policy.retainIndefinitely) {
        continue; // No retention policy set or indefinite retention
      }

      const cutoffDate = new Date(Date.now() - days * 24 * 60 * 60 * 1000);

      // Find files to soft delete
      const filesToDelete = await detectionFileRepository.find({
        where: [
          { fileType: type, cameraId: cameraId === 'global' ? 'cam1' : cameraId, isDeleted: false },
          { fileType: type, cameraId: cameraId === 'global' ? 'cam2' : cameraId, isDeleted: false }
        ]
      });

      // Filter by cutoff date
      const oldFiles = filesToDelete.filter(f => {
        const captureDate = f.captureTimestamp || f.createdAt;
        return captureDate < cutoffDate;
      });

      if (oldFiles.length > 0) {
        console.log(`[RetentionCleanup] Soft deleting ${oldFiles.length} ${type} files for ${cameraId}`);

        // Mark as deleted
        await detectionFileRepository.update(
          oldFiles.map(f => f.fileUuid),
          { isDeleted: true, deletedAt: new Date() }
        );

        result.softDeleted += oldFiles.length;
        result.processed += oldFiles.length;
      }
    }
  }

  /**
   * Archive soft-deleted files past grace period (7 days)
   */
  private async archiveSoftDeletedFiles(policy: RetentionPolicy, result: CleanupResult): Promise<void> {
    const gracePeriodDays = 7;
    const gracePeriodCutoff = new Date(Date.now() - gracePeriodDays * 24 * 60 * 60 * 1000);

    const cameraId = policy.camera;

    // Find soft-deleted files past grace period
    const filesToArchive = await detectionFileRepository.find({
      where: [
        { cameraId: cameraId === 'global' ? 'cam1' : cameraId, isDeleted: true, isArchived: false },
        { cameraId: cameraId === 'global' ? 'cam2' : cameraId, isDeleted: true, isArchived: false }
      ]
    });

    // Filter by grace period
    const oldSoftDeleted = filesToArchive.filter(f => {
      return f.deletedAt && f.deletedAt < gracePeriodCutoff;
    });

    if (oldSoftDeleted.length === 0) {
      return;
    }

    console.log(`[RetentionCleanup] Archiving ${oldSoftDeleted.length} files for ${cameraId}`);

    for (const file of oldSoftDeleted) {
      try {
        await this.archiveFile(file);
        result.archived++;
        result.processed++;
      } catch (error) {
        console.error(`[RetentionCleanup] Error archiving file ${file.fileUuid}:`, error);
        result.errors++;
      }
    }
  }

  /**
   * Move file to archive directory
   */
  private async archiveFile(file: DetectionFile): Promise<void> {
    const currentPath = file.storagePath;
    const archiveDir = path.join(process.env.ARCHIVE_DIR || '/app/data/detections/archive', this.getYearMonth(file.createdAt));
    const archivePath = path.join(archiveDir, path.basename(currentPath));

    // Create archive directory if needed
    await fs.mkdir(archiveDir, { recursive: true });

    // Verify file exists
    try {
      await fs.access(currentPath);
    } catch {
      console.warn(`[RetentionCleanup] File not found, skipping: ${currentPath}`);
      // Mark as archived anyway (file already gone)
      await detectionFileRepository.update(file.fileUuid, {
        isArchived: true,
        archivedAt: new Date(),
        storagePath: archivePath
      });
      return;
    }

    // Verify integrity before archiving
    const isValid = await this.verifyFileIntegrity(file);
    if (!isValid) {
      console.warn(`[RetentionCleanup] File integrity check failed, skipping: ${currentPath}`);
      result.errors = (result.errors || 0) + 1;
      return;
    }

    // Move file to archive
    await fs.rename(currentPath, archivePath);

    // Update database
    await detectionFileRepository.update(file.fileUuid, {
      isArchived: true,
      archivedAt: new Date(),
      storagePath: archivePath
    });
  }

  /**
   * Hard delete archived files past archive retention
   */
  private async hardDeleteArchivedFiles(policy: RetentionPolicy, result: CleanupResult): Promise<void> {
    const archiveRetentionDays = policy.archivesDays || 90;
    const archiveCutoff = new Date(Date.now() - archiveRetentionDays * 24 * 60 * 60 * 1000);

    const cameraId = policy.camera;

    // Find archived files past retention
    const filesToDelete = await detectionFileRepository.find({
      where: [
        { cameraId: cameraId === 'global' ? 'cam1' : cameraId, isArchived: true },
        { cameraId: cameraId === 'global' ? 'cam2' : cameraId, isArchived: true }
      ]
    });

    // Filter by archive cutoff date
    const oldArchived = filesToDelete.filter(f => {
      return f.archivedAt && f.archivedAt < archiveCutoff;
    });

    if (oldArchived.length === 0) {
      return;
    }

    console.log(`[RetentionCleanup] Hard deleting ${oldArchived.length} archived files for ${cameraId}`);

    for (const file of oldArchived) {
      try {
        await this.hardDeleteFile(file);
        result.hardDeleted++;
        result.processed++;
      } catch (error) {
        console.error(`[RetentionCleanup] Error hard deleting file ${file.fileUuid}:`, error);
        result.errors++;
      }
    }
  }

  /**
   * Permanently delete file and database record
   */
  private async hardDeleteFile(file: DetectionFile): Promise<void> {
    // Delete file from filesystem
    try {
      await fs.unlink(file.storagePath);
    } catch (error) {
      // File might not exist, that's okay
      console.warn(`[RetentionCleanup] File not found for deletion: ${file.storagePath}`);
    }

    // Delete database record
    await detectionFileRepository.delete(file.fileUuid);
  }

  /**
   * Verify file integrity using SHA-256 hash
   */
  private async verifyFileIntegrity(file: DetectionFile): Promise<boolean> {
    try {
      const fileBuffer = await fs.readFile(file.storagePath);
      const hash = createHash('sha256').update(fileBuffer).digest('hex');
      return hash === file.fileHash;
    } catch (error) {
      console.error(`[RetentionCleanup] Integrity check error for ${file.storagePath}:`, error);
      return false;
    }
  }

  /**
   * Vacuum database to reclaim space
   */
  private async vacuumDatabase(): Promise<void> {
    try {
      await AppDataSource.query('VACUUM ANALYZE detection_files');
      console.log('[RetentionCleanup] Database vacuum completed');
    } catch (error) {
      console.error('[RetentionCleanup] Vacuum error:', error);
      // Don't fail cleanup if vacuum fails
    }
  }

  /**
   * Log cleanup operation to audit
   */
  private async logCleanupToAudit(result: CleanupResult): Promise<void> {
    try {
      await AppDataSource.query(
        `INSERT INTO audit_logs (user_id, action, details, ip_address)
         VALUES ($1, $2, $3, $4)`,
        [
          null, // System operation
          'retention_cleanup',
          JSON.stringify(result),
          'system'
        ]
      );
    } catch (error) {
      console.error('[RetentionCleanup] Audit log error:', error);
    }
  }

  /**
   * Get year-month string for archive directory
   */
  private getYearMonth(date: Date): string {
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, '0');
    return `${year}-${month}`;
  }

  /**
   * Get cleanup status for UI
   */
  async getCleanupStatus(): Promise<{
    lastCleanup: CleanupResult | null;
    nextCleanup: Date;
  }> {
    // Get last cleanup from audit logs
    const lastCleanupResult = await AppDataSource.query(
      `SELECT details FROM audit_logs
       WHERE action = 'retention_cleanup'
       ORDER BY created_at DESC
       LIMIT 1`
    );

    const lastCleanup = lastCleanupResult.length > 0
      ? JSON.parse(lastCleanupResult[0].details) as CleanupResult
      : null;

    // Next cleanup is at 3 AM tomorrow
    const tomorrow = new Date();
    tomorrow.setDate(tomorrow.getDate() + 1);
    tomorrow.setHours(3, 0, 0, 0);

    return {
      lastCleanup,
      nextCleanup: tomorrow
    };
  }
}

export const retentionCleanupService = new RetentionCleanupService();
```

### Step 4: API Routes for Manual Cleanup

**File:** `server/src/routes/cleanupRoutes.ts`

```typescript
import express from 'express';
import { retentionCleanupService } from '../services/retentionCleanupService.js';

const router = express.Router();

// POST /api/storage/cleanup - Trigger manual cleanup
router.post('/', async (req, res) => {
  try {
    // Check admin permissions
    if (req.user?.role !== 'admin') {
      return res.status(403).json({ error: 'Admin access required' });
    }

    console.log('[API] Manual cleanup triggered by', req.user?.id);

    const result = await retentionCleanupService.enforcePolicies();

    res.json({
      message: 'Cleanup completed',
      result
    });
  } catch (error) {
    console.error('[API] Manual cleanup error:', error);
    res.status(500).json({ error: 'Cleanup failed' });
  }
});

// GET /api/storage/cleanup/status - Get cleanup status
router.get('/status', async (req, res) => {
  try {
    const status = await retentionCleanupService.getCleanupStatus();
    res.json(status);
  } catch (error) {
    console.error('[API] Cleanup status error:', error);
    res.status(500).json({ error: 'Failed to get status' });
  }
});

export default router;
```
