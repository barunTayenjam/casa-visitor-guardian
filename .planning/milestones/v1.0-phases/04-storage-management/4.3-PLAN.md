### Step 1: Storage Statistics Service

**File:** `server/src/services/storageStatsService.ts`

```typescript
import { AppDataSource } from '../database.js';
import { DetectionFile } from '../models/DetectionFile.js';
import { StorageStats } from '../models/StorageStats.js';
import * as fs from 'fs/promises';
import * as path from 'path';

const detectionFileRepository = AppDataSource.getRepository(DetectionFile);
const storageStatsRepository = AppDataSource.getRepository(StorageStats);

export interface StorageUsage {
  totalBytes: number;
  totalFiles: number;
  byType: Record<string, { bytes: number; files: number }>;
  byCamera: Record<string, { bytes: number; files: number }>;
}

export interface StorageProjection {
  daysUntilExhaustion: number | null;
  exhaustionDate: Date | null;
  dailyGrowthRate: number; // bytes per day
  projectedUsage: Array<{ date: Date; bytes: number }>;
}

export class StorageStatsService {
  /**
   * Calculate and store hourly storage statistics
   */
  async calculateAndStoreStats(): Promise<void> {
    try {
      console.log('[StorageStats] Calculating hourly statistics...');

      const now = new Date();
      const today = new Date(now.getFullYear(), now.getMonth(), now.getDate());

      // Get stats by file type
      const typeStats = await detectionFileRepository
        .createQueryBuilder('file')
        .select('file.fileType', 'fileType')
        .addSelect('COUNT(*)', 'totalFiles')
        .addSelect('SUM(file.fileSize)', 'totalSize')
        .where('file.createdAt >= :today', { today })
        .groupBy('file.fileType')
        .getRawMany();

      // Get stats by camera
      const cameraStats = await detectionFileRepository
        .createQueryBuilder('file')
        .select('file.cameraId', 'cameraId')
        .addSelect('COUNT(*)', 'totalFiles')
        .addSelect('SUM(file.fileSize)', 'totalSize')
        .where('file.createdAt >= :today', { today })
        .groupBy('file.cameraId')
        .getRawMany();

      // Store stats
      for (const typeStat of typeStats) {
        await storageStatsRepository.save({
          statDate: now,
          fileType: typeStat.fileType,
          totalFiles: parseInt(typeStat.totalFiles),
          totalSize: parseInt(typeStat.totalSize) || 0,
          newFiles: parseInt(typeStat.totalFiles),
          deletedFiles: 0,
          archivedFiles: 0,
          calculatedAt: now
        });
      }

      for (const cameraStat of cameraStats) {
        await storageStatsRepository.save({
          statDate: now,
          fileType: cameraStat.cameraId,
          totalFiles: parseInt(cameraStat.totalFiles),
          totalSize: parseInt(cameraStat.totalSize) || 0,
          newFiles: parseInt(cameraStat.totalFiles),
          deletedFiles: 0,
          archivedFiles: 0,
          calculatedAt: now
        });
      }

      console.log('[StorageStats] Hourly statistics calculated and stored');
    } catch (error) {
      console.error('[StorageStats] Error calculating statistics:', error);
      throw error;
    }
  }

  /**
   * Get current storage usage
   */
  async getCurrentUsage(): Promise<StorageUsage> {
    // Get total stats from database
    const totalStats = await detectionFileRepository
      .createQueryBuilder('file')
      .select('COUNT(*)', 'totalFiles')
      .addSelect('SUM(file.fileSize)', 'totalSize')
      .where('file.isDeleted = :isDeleted', { isDeleted: false })
      .getRawOne();

    const totalBytes = parseInt(totalStats?.totalSize) || 0;
    const totalFiles = parseInt(totalStats?.totalFiles) || 0;

    // Get by type
    const typeStats = await detectionFileRepository
      .createQueryBuilder('file')
      .select('file.fileType', 'fileType')
      .addSelect('COUNT(*)', 'totalFiles')
      .addSelect('SUM(file.fileSize)', 'totalSize')
      .where('file.isDeleted = :isDeleted', { isDeleted: false })
      .groupBy('file.fileType')
      .getRawMany();

    const byType: Record<string, { bytes: number; files: number }> = {};
    for (const stat of typeStats) {
      byType[stat.fileType] = {
        bytes: parseInt(stat.totalSize) || 0,
        files: parseInt(stat.totalFiles) || 0
      };
    }

    // Get by camera
    const cameraStats = await detectionFileRepository
      .createQueryBuilder('file')
      .select('file.cameraId', 'cameraId')
      .addSelect('COUNT(*)', 'totalFiles')
      .addSelect('SUM(file.fileSize)', 'totalSize')
      .where('file.isDeleted = :isDeleted', { isDeleted: false })
      .groupBy('file.cameraId')
      .getRawMany();

    const byCamera: Record<string, { bytes: number; files: number }> = {};
    for (const stat of cameraStats) {
      byCamera[stat.cameraId] = {
        bytes: parseInt(stat.totalSize) || 0,
        files: parseInt(stat.totalFiles) || 0
      };
    }

    return {
      totalBytes,
      totalFiles,
      byType,
      byCamera
    };
  }

  /**
   * Get historical storage stats for trends
   */
  async getHistoricalStats(days: number = 30): Promise<Array<{
    date: Date;
    totalBytes: number;
    totalFiles: number;
  }>> {
    const cutoffDate = new Date(Date.now() - days * 24 * 60 * 60 * 1000);

    const stats = await storageStatsRepository
      .createQueryBuilder('stats')
      .select('DATE(stats.statDate)', 'date')
      .addSelect('SUM(stats.totalSize)', 'totalBytes')
      .addSelect('SUM(stats.totalFiles)', 'totalFiles')
      .where('stats.statDate >= :cutoff', { cutoff: cutoffDate })
      .groupBy('DATE(stats.statDate)')
      .orderBy('DATE(stats.statDate)', 'ASC')
      .getRawMany();

    return stats.map(s => ({
      date: new Date(s.date),
      totalBytes: parseInt(s.totalBytes) || 0,
      totalFiles: parseInt(s.totalFiles) || 0
    }));
  }

  /**
   * Calculate storage projection (time to exhaustion)
   */
  async getProjection(): Promise<StorageProjection> {
    const maxStorageGB = parseInt(process.env.MAX_STORAGE_GB || '100');
    const maxBytes = maxStorageGB * 1024 * 1024 * 1024;

    // Get last 30 days of stats
    const historical = await this.getHistoricalStats(30);

    if (historical.length < 2) {
      return {
        daysUntilExhaustion: null,
        exhaustionDate: null,
        dailyGrowthRate: 0,
        projectedUsage: []
      };
    }

    // Calculate daily growth rate (linear regression)
    const oldest = historical[0];
    const newest = historical[historical.length - 1];
    const daysDiff = (newest.date.getTime() - oldest.date.getTime()) / (1000 * 60 * 60 * 24);
    const bytesDiff = newest.totalBytes - oldest.totalBytes;
    const dailyGrowthRate = bytesDiff / daysDiff;

    // Current usage
    const current = await this.getCurrentUsage();
    const currentBytes = current.totalBytes;

    // Project usage
    const projectedUsage: Array<{ date: Date; bytes: number }> = [];
    let projectedBytes = currentBytes;
    let projectedDate = new Date();

    for (let i = 1; i <= 90; i++) {
      projectedDate = new Date();
      projectedDate.setDate(projectedDate.getDate() + i);
      projectedBytes += dailyGrowthRate;

      projectedUsage.push({
        date: projectedDate,
        bytes: Math.max(0, projectedBytes)
      });

      if (projectedBytes >= maxBytes) {
        return {
          daysUntilExhaustion: i,
          exhaustionDate: projectedDate,
          dailyGrowthRate,
          projectedUsage
        };
      }
    }

    return {
      daysUntilExhaustion: null,
      exhaustionDate: null,
      dailyGrowthRate,
      projectedUsage
    };
  }

  /**
   * Get cleanup history from audit logs
   */
  async getCleanupHistory(limit: number = 10): Promise<Array<{
    id: string;
    timestamp: Date;
    result: any;
  }>> {
    const history = await AppDataSource.query(
      `SELECT id, created_at as timestamp, details as result
       FROM audit_logs
       WHERE action = 'retention_cleanup'
       ORDER BY created_at DESC
       LIMIT $1`,
      [limit]
    );

    return history.map(h => ({
      id: h.id,
      timestamp: new Date(h.timestamp),
      result: typeof h.result === 'string' ? JSON.parse(h.result) : h.result
    }));
  }

  /**
   * Get storage health status
   */
  async getHealthStatus(): Promise<{
    status: 'healthy' | 'warning' | 'critical';
    usagePercent: number;
    usedBytes: number;
    totalBytes: number;
    alertMessage?: string;
  }> {
    const usage = await this.getCurrentUsage();
    const maxStorageGB = parseInt(process.env.MAX_STORAGE_GB || '100');
    const totalBytes = maxStorageGB * 1024 * 1024 * 1024;
    const usagePercent = (usage.totalBytes / totalBytes) * 100;

    let status: 'healthy' | 'warning' | 'critical' = 'healthy';
    let alertMessage: string | undefined;

    if (usagePercent >= 95) {
      status = 'critical';
      alertMessage = `Storage critically high: ${usagePercent.toFixed(1)}% used. Immediate action required.`;
    } else if (usagePercent >= 80) {
      status = 'warning';
      alertMessage = `Storage usage at ${usagePercent.toFixed(1)}%. Consider cleanup or expansion.`;
    }

    return {
      status,
      usagePercent,
      usedBytes: usage.totalBytes,
      totalBytes,
      alertMessage
    };
  }
}

export const storageStatsService = new StorageStatsService();
```

```typescript
import { Entity, PrimaryColumn, Column, Index, CreateDateColumn } from 'typeorm';

@Entity('storage_stats')
@Index(['statDate'])

  @PrimaryColumn('uuid')

  @Column({ type: 'timestamp with time zone' })
  statDate!: Date;

  @Column({ type: 'bigint' })

  @Column({ type: 'integer' })
  totalFiles!: number;

  @Column({ type: 'integer', default: 0 })
  newFiles!: number;

  @Column({ type: 'integer', default: 0 })
  deletedFiles!: number;

  @Column({ type: 'integer', default: 0 })
  archivedFiles!: number;

  @CreateDateColumn({ name: 'calculated_at' })
```

```typescript
import React, { useEffect, useState } from 'react';
import { PieChart, Pie, Cell, BarChart, Bar, LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer } from 'recharts';

interface StorageUsage {
  totalBytes: number;
  totalFiles: number;
  byType: Record<string, { bytes: number; files: number }>;
  byCamera: Record<string, { bytes: number; files: number }>;
}

interface HealthStatus {
  status: 'healthy' | 'warning' | 'critical';
  usagePercent: number;
  usedBytes: number;
  totalBytes: number;
  alertMessage?: string;
}

export const StorageDashboard: React.FC = () => {
  const [usage, setUsage] = useState<StorageUsage | null>(null);
  const [health, setHealth] = useState<HealthStatus | null>(null);
  const [history, setHistory] = useState<any[]>([]);
  const [projection, setProjection] = useState<any | null>(null);
  const [cleanupHistory, setCleanupHistory] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  const fetchData = async () => {
    try {
      const [usageRes, healthRes, historyRes, projectionRes, cleanupRes] = await Promise.all([
        fetch('/api/storage/stats'),
        fetch('/api/storage/stats/health'),
        fetch('/api/storage/stats/history?days=30'),
        fetch('/api/storage/stats/projection'),
        fetch('/api/storage/stats/cleanup-history?limit=5')
      ]);

      const [usageData, healthData, historyData, projectionData, cleanupData] = await Promise.all([
        usageRes.json(),
        healthRes.json(),
        historyRes.json(),
        projectionRes.json(),
        cleanupRes.json()
      ]);

      setUsage(usageData.usage);
      setHealth(healthData.health);
      setHistory(historyData.history);
      setProjection(projectionData.projection);
      setCleanupHistory(cleanupData.history);
    } catch (error) {
      console.error('Error fetching storage data:', error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
    const interval = setInterval(fetchData, 60000); // Auto-refresh every 60s
    return () => clearInterval(interval);
  }, []);

  if (loading) {
    return <div className="p-6">Loading storage dashboard...</div>;
  }

  const COLORS = ['#0088FE', '#00C49F', '#FFBB28', '#FF8042', '#8884D8'];

  // Prepare pie chart data (by type)
  const pieData = usage?.byType
    ? Object.entries(usage.byType).map(([type, data]) => ({
        name: type,
        value: data.bytes,
        files: data.files
      }))
    : [];

  // Prepare bar chart data (by camera)
  const barData = usage?.byCamera
    ? Object.entries(usage.byCamera).map(([camera, data]) => ({
        camera,
        bytes: data.bytes,
        files: data.files
      }))
    : [];

  // Prepare line chart data (history)
  const lineData = history.map(h => ({
    date: new Date(h.date).toLocaleDateString(),
    bytes: h.totalBytes,
    files: h.totalFiles
  }));

  return (
    <div className="p-6 space-y-6">
      <h1 className="text-3xl font-bold">Storage Dashboard</h1>

      {/* Alert Banner */}
      {health?.alertMessage && (
        <div className={`p-4 rounded-lg ${health.status === 'critical' ? 'bg-red-100 text-red-800' : 'bg-yellow-100 text-yellow-800'}`}>
          <strong>Alert:</strong> {health.alertMessage}
        </div>
      )}

      {/* Storage Gauge */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <div className="bg-white p-6 rounded-lg shadow">
          <h3 className="text-lg font-semibold mb-2">Total Storage Used</h3>
          <div className="text-4xl font-bold">{((health?.usedBytes || 0) / (1024 ** 3)).toFixed(2)} GB</div>
          <div className="text-sm text-gray-500">of {((health?.totalBytes || 0) / (1024 ** 3)).toFixed(2)} GB</div>
          <div className="mt-4 h-2 bg-gray-200 rounded-full overflow-hidden">
            <div
              className={`h-full ${health?.status === 'critical' ? 'bg-red-500' : health?.status === 'warning' ? 'bg-yellow-500' : 'bg-green-500'}`}
              style={{ width: `${health?.usagePercent || 0}%` }}
            />
          </div>
          <div className="text-sm mt-2">{health?.usagePercent.toFixed(1)}% used</div>
        </div>

        <div className="bg-white p-6 rounded-lg shadow">
          <h3 className="text-lg font-semibold mb-2">Total Files</h3>
          <div className="text-4xl font-bold">{usage?.totalFiles.toLocaleString()}</div>
          <div className="text-sm text-gray-500">files indexed</div>
        </div>

        <div className="bg-white p-6 rounded-lg shadow">
          <h3 className="text-lg font-semibold mb-2">Projection</h3>
          {projection?.daysUntilExhaustion ? (
            <>
              <div className="text-4xl font-bold text-red-600">{projection.daysUntilExhaustion} days</div>
              <div className="text-sm text-gray-500">until exhaustion</div>
              <div className="text-xs text-gray-400 mt-2">
                {new Date(projection.exhaustionDate).toLocaleDateString()}
              </div>
            </>
          ) : (
            <>
              <div className="text-4xl font-bold text-green-600">∞</div>
              <div className="text-sm text-gray-500">no exhaustion risk</div>
              <div className="text-xs text-gray-400 mt-2">
                +{Math.round(projection?.dailyGrowthRate / (1024 ** 2))} MB/day
              </div>
            </>
          )}
        </div>
      </div>

      {/* Charts */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        {/* Storage by Type (Pie) */}
        <div className="bg-white p-6 rounded-lg shadow">
          <h3 className="text-lg font-semibold mb-4">Storage by File Type</h3>
          <ResponsiveContainer width="100%" height={300}>
            <PieChart>
              <Pie
                data={pieData}
                cx="50%"
                cy="50%"
                labelLine={false}
                label={(entry) => `${entry.name} ${(entry.value / (1024 ** 3)).toFixed(2)} GB`}
                outerRadius={80}
                fill="#8884d8"
                dataKey="value"
              >
                {pieData.map((entry, index) => (
                  <Cell key={`cell-${index}`} fill={COLORS[index % COLORS.length]} />
                ))}
              </Pie>
              <Tooltip formatter={(value: number) => `${(value / (1024 ** 3)).toFixed(2)} GB`} />
            </PieChart>
          </ResponsiveContainer>
        </div>

        {/* Storage by Camera (Bar) */}
        <div className="bg-white p-6 rounded-lg shadow">
          <h3 className="text-lg font-semibold mb-4">Storage by Camera</h3>
          <ResponsiveContainer width="100%" height={300}>
            <BarChart data={barData}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis dataKey="camera" />
              <YAxis />
              <Tooltip formatter={(value: number) => `${(value / (1024 ** 3)).toFixed(2)} GB`} />
              <Legend />
              <Bar dataKey="bytes" fill="#8884d8" name="Storage (GB)" />
            </BarChart>
          </ResponsiveContainer>
        </div>
      </div>

      {/* Historical Trend (Line) */}
      <div className="bg-white p-6 rounded-lg shadow">
        <h3 className="text-lg font-semibold mb-4">30-Day Storage Trend</h3>
        <ResponsiveContainer width="100%" height={300}>
          <LineChart data={lineData}>
            <CartesianGrid strokeDasharray="3 3" />
            <XAxis dataKey="date" />
            <YAxis />
            <Tooltip formatter={(value: number) => `${(value / (1024 ** 3)).toFixed(2)} GB`} />
            <Legend />
            <Line type="monotone" dataKey="bytes" stroke="#8884d8" name="Storage (GB)" />
          </LineChart>
        </ResponsiveContainer>
      </div>

      {/* Cleanup History */}
      <div className="bg-white p-6 rounded-lg shadow">
        <h3 className="text-lg font-semibold mb-4">Recent Cleanup Jobs</h3>
        {cleanupHistory.length === 0 ? (
          <p className="text-gray-500">No cleanup history yet</p>
        ) : (
          <div className="space-y-2">
            {cleanupHistory.map((job) => (
              <div key={job.id} className="p-3 bg-gray-50 rounded">
                <div className="flex justify-between items-center">
                  <span className="font-medium">{new Date(job.timestamp).toLocaleString()}</span>
                  <span className={`px-2 py-1 rounded text-xs ${job.result.success ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'}`}>
                    {job.result.success ? 'Success' : 'Failed'}
                  </span>
                </div>
                <div className="text-sm text-gray-600 mt-1">
                  Processed: {job.result.processed} |
                  Soft deleted: {job.result.softDeleted} |
                  Archived: {job.result.archived} |
                  Hard deleted: {job.result.hardDeleted} |
                  Duration: {job.result.duration}ms
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
};
```
