# Plan 4.1: Define Retention Policy Schema — Configurable Periods Per Type

**Phase:** 4 - Storage Management
**Status:** Pending
**Priority:** P0 (Critical)
**Dependencies:** None
**Requirements:** STOR-01, STOR-02

## Objective

Define and implement a comprehensive retention policy schema that supports configurable retention periods for different data types (events, images, videos, logs) with per-camera override capabilities and policy enforcement validation.

## Success Criteria

1. Retention policy schema supports all data types (alerts, detections, previews, snapshots, events, logs)
2. Policies are configurable per camera with global defaults
3. Policy validation ensures no conflicts or invalid values
4. Database schema includes proper constraints and indexes
5. API endpoints for policy management (create, read, update, delete)
6. "Indefinite retention" flag supported for critical data
7. Policies apply to both file-based and database-stored data

## Current State Analysis

**Database (PostgreSQL):**
- `retention_policies` table exists from migration 007
- Table has columns: camera, alerts_days, detections_days, previews_days, snapshots_days, events_days, retain_indefinitely
- **0 rows in table** - schema exists but never populated
- No policy validation logic
- No API endpoints for management

**Configuration (.env):**
- `RETENTION_DAYS=30` defined but not enforced
- No per-camera configuration
- No differentiation between data types

**Issues:**
- Schema exists but completely unused
- No default policies seeded
- No validation or constraint checking
- No way to manage policies via API
- No audit trail for policy changes

```typescript
import { Entity, PrimaryColumn, Column, Index, CreateDateColumn, UpdateDateColumn, ManyToOne, JoinColumn } from 'typeorm';
import { User } from './User.js';

@Entity('retention_policies')
@Index(['camera'])
@Index(['isActive'])
export class RetentionPolicy {
  @PrimaryColumn('uuid')
  id!: string;

  @Column({ type: 'text' })
  camera!: string; // 'global' or camera ID (e.g., 'cam1', 'cam2')

  // Retention periods (days)
  @Column({ type: 'integer', nullable: true })
  alertsDays?: number; // 30 days default

  @Column({ type: 'integer', nullable: true })
  detectionsDays?: number; // 7 days default

  @Column({ type: 'integer', nullable: true })
  previewsDays?: number; // 7 days default

  @Column({ type: 'integer', nullable: true })
  snapshotsDays?: number; // 30 days default

  @Column({ type: 'integer', nullable: true })
  eventsDays?: number; // 30 days default

  @Column({ type: 'integer', nullable: true })
  logsDays?: number; // 7 days default

  @Column({ type: 'integer', nullable: true })
  batchResultsDays?: number; // 7 days default

  @Column({ type: 'integer', nullable: true })
  tempFilesDays?: number; // 1 day default

  @Column({ type: 'integer', nullable: true })
  archivesDays?: number; // 90 days default

  // Special flag
  @Column({ type: 'boolean', default: false })
  retainIndefinitely!: boolean; // If true, never delete

  // Metadata
  @Column({ type: 'boolean', default: true })
  isActive!: boolean;

  @Column({ type: 'text', nullable: true })
  description?: string;

  @ManyToOne(() => User)
  @JoinColumn({ name: 'created_by' })
  createdBy?: User;

  @CreateDateColumn({ name: 'created_at' })
  createdAt!: Date;

  @UpdateDateColumn({ name: 'updated_at' })
  updatedAt!: Date;

  // Helper method to get retention for a specific file type
  getRetentionForType(fileType: string): number | null {
    if (this.retainIndefinitely) {
      return null; // Never delete
    }

    switch (fileType) {
      case 'event_face':
      case 'event_motion':
        return this.detectionsDays ?? null;
      case 'event_alert':
        return this.alertsDays ?? null;
      case 'snapshot':
        return this.snapshotsDays ?? null;
      case 'preview':
        return this.previewsDays ?? null;
      case 'batch_result':
        return this.batchResultsDays ?? null;
      case 'temp':
        return this.tempFilesDays ?? null;
      case 'archive':
        return this.archivesDays ?? null;
      default:
        return null;
    }
  }

  // Helper method to check if file should be deleted
  shouldDelete(fileType: string, fileAgeDays: number): boolean {
    if (this.retainIndefinitely) {
      return false;
    }

    const retentionDays = this.getRetentionForType(fileType);
    if (retentionDays === null) {
      return false; // No retention policy set
    }

    return fileAgeDays > retentionDays;
  }
}
```

### Step 4: Retention Policy Service

**File:** `server/src/services/retentionPolicyService.ts`

```typescript
import { AppDataSource } from '../database.js';
import { RetentionPolicy, RetentionPolicyAudit } from '../models/RetentionPolicy.js';

const policyRepository = AppDataSource.getRepository(RetentionPolicy);
const auditRepository = AppDataSource.getRepository(RetentionPolicyAudit);

export class RetentionPolicyService {
  /**
   * Get policy for a specific camera (falls back to global)
   */
  async getPolicyForCamera(cameraId: string): Promise<RetentionPolicy> {
    // Try camera-specific policy first
    const cameraPolicy = await policyRepository.findOne({
      where: { camera: cameraId, isActive: true }
    });

    if (cameraPolicy) {
      return cameraPolicy;
    }

    // Fall back to global policy
    const globalPolicy = await policyRepository.findOne({
      where: { camera: 'global', isActive: true }
    });

    if (!globalPolicy) {
      throw new Error('No retention policy found (not even global)');
    }

    return globalPolicy;
  }

  /**
   * Get all policies
   */
  async getAllPolicies(): Promise<RetentionPolicy[]> {
    return await policyRepository.find({
      order: { camera: 'ASC' }
    });
  }

  /**
   * Get policy by ID
   */
  async getPolicyById(id: string): Promise<RetentionPolicy | null> {
    return await policyRepository.findOne({ where: { id } });
  }

  /**
   * Create or update policy
   */
  async upsertPolicy(
    cameraId: string,
    policyData: Partial<RetentionPolicy>,
    userId: string,
    reason?: string
  ): Promise<RetentionPolicy> {
    // Check if policy exists
    const existingPolicy = await policyRepository.findOne({
      where: { camera: cameraId }
    });

    if (existingPolicy) {
      // Update existing policy
      const oldValues = { ...existingPolicy };

      // Update fields
      Object.assign(existingPolicy, policyData);

      const updatedPolicy = await policyRepository.save(existingPolicy);

      // Audit log
      await auditRepository.save({
        policyId: updatedPolicy.id,
        changedBy: userId,
        oldValues,
        newValues: { ...updatedPolicy },
        changeType: 'update',
        reason
      });

      return updatedPolicy;
    } else {
      // Create new policy
      const newPolicy = policyRepository.create({
        id: crypto.randomUUID(),
        camera: cameraId,
        ...policyData
      });

      const savedPolicy = await policyRepository.save(newPolicy);

      // Audit log
      await auditRepository.save({
        policyId: savedPolicy.id,
        changedBy: userId,
        oldValues: {},
        newValues: { ...savedPolicy },
        changeType: 'create',
        reason
      });

      return savedPolicy;
    }
  }

  /**
   * Delete policy (soft delete by setting isActive = false)
   */
  async deletePolicy(id: string, userId: string, reason?: string): Promise<void> {
    const policy = await policyRepository.findOne({ where: { id } });

    if (!policy) {
      throw new Error(`Policy not found: ${id}`);
    }

    // Don't allow deleting global policy
    if (policy.camera === 'global') {
      throw new Error('Cannot delete global retention policy');
    }

    // Soft delete
    const oldValues = { ...policy };
    policy.isActive = false;

    await policyRepository.save(policy);

    // Audit log
    await auditRepository.save({
      policyId: id,
      changedBy: userId,
      oldValues,
      newValues: { isActive: false },
      changeType: 'delete',
      reason
    });
  }

  /**
   * Validate policy data
   */
  validatePolicyData(data: Partial<RetentionPolicy>): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    // Check at least one retention type is set
    const hasRetention =
      data.alertsDays !== undefined ||
      data.detectionsDays !== undefined ||
      data.previewsDays !== undefined ||
      data.snapshotsDays !== undefined ||
      data.eventsDays !== undefined ||
      data.logsDays !== undefined ||
      data.batchResultsDays !== undefined ||
      data.tempFilesDays !== undefined ||
      data.archivesDays !== undefined;

    if (!hasRetention) {
      errors.push('At least one retention period must be specified');
    }

    // Check all retention periods are positive
    const retentionFields = [
      'alertsDays', 'detectionsDays', 'previewsDays', 'snapshotsDays',
      'eventsDays', 'logsDays', 'batchResultsDays', 'tempFilesDays', 'archivesDays'
    ];

    for (const field of retentionFields) {
      const value = data[field as keyof RetentionPolicy] as number | undefined;
      if (value !== undefined && value !== null && value <= 0) {
        errors.push(`${field} must be greater than 0`);
      }
    }

    // Check indefinite retention logic
    if (data.retainIndefinitely === true) {
      if (data.alertsDays !== null || data.detectionsDays !== null) {
        errors.push('Cannot set both retainIndefinitely=true and specific retention periods');
      }
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }

  /**
   * Get audit log for a policy
   */
  async getPolicyAudit(policyId: string): Promise<RetentionPolicyAudit[]> {
    return await auditRepository.find({
      where: { policyId },
      order: { changedAt: 'DESC' }
    });
  }
}

export const retentionPolicyService = new RetentionPolicyService();
```

### Step 5: API Routes

**File:** `server/src/routes/retentionPolicyRoutes.ts`

```typescript
import express from 'express';
import { retentionPolicyService } from '../services/retentionPolicyService.js';

const router = express.Router();

// GET /api/retention-policies - List all policies
router.get('/', async (req, res) => {
  try {
    const policies = await retentionPolicyService.getAllPolicies();
    res.json({ policies });
  } catch (error) {
    console.error('Error fetching policies:', error);
    res.status(500).json({ error: 'Failed to fetch policies' });
  }
});

// GET /api/retention-policies/camera/:cameraId - Get policy for camera
router.get('/camera/:cameraId', async (req, res) => {
  try {
    const { cameraId } = req.params;
    const policy = await retentionPolicyService.getPolicyForCamera(cameraId);
    res.json({ policy });
  } catch (error) {
    console.error('Error fetching policy:', error);
    res.status(404).json({ error: 'Policy not found' });
  }
});

// GET /api/retention-policies/:id - Get policy by ID
router.get('/:id', async (req, res) => {
  try {
    const { id } = req.params;
    const policy = await retentionPolicyService.getPolicyById(id);

    if (!policy) {
      return res.status(404).json({ error: 'Policy not found' });
    }

    res.json({ policy });
  } catch (error) {
    console.error('Error fetching policy:', error);
    res.status(500).json({ error: 'Failed to fetch policy' });
  }
});

// PUT /api/retention-policies/:cameraId - Create or update policy
router.put('/:cameraId', async (req, res) => {
  try {
    const { cameraId } = req.params;
    const { alertsDays, detectionsDays, previewsDays, snapshotsDays, eventsDays,
            logsDays, batchResultsDays, tempFilesDays, archivesDays,
            retainIndefinitely, description } = req.body;

    // Validate policy data
    const validation = retentionPolicyService.validatePolicyData({
      alertsDays, detectionsDays, previewsDays, snapshotsDays, eventsDays,
      logsDays, batchResultsDays, tempFilesDays, archivesDays, retainIndefinitely
    });

    if (!validation.valid) {
      return res.status(400).json({ error: 'Invalid policy data', details: validation.errors });
    }

    // Get user ID from auth (placeholder)
    const userId = req.user?.id || 'system';

    // Upsert policy
    const policy = await retentionPolicyService.upsertPolicy(
      cameraId,
      { alertsDays, detectionsDays, previewsDays, snapshotsDays, eventsDays,
        logsDays, batchResultsDays, tempFilesDays, archivesDays,
        retainIndefinitely, description },
      userId,
      req.body.reason
    );

    res.json({ policy, message: 'Policy updated successfully' });
  } catch (error) {
    console.error('Error updating policy:', error);
    res.status(500).json({ error: 'Failed to update policy' });
  }
});

// DELETE /api/retention-policies/:id - Delete policy
router.delete('/:id', async (req, res) => {
  try {
    const { id } = req.params;
    const userId = req.user?.id || 'system';

    await retentionPolicyService.deletePolicy(id, userId, req.body.reason);

    res.json({ message: 'Policy deleted successfully' });
  } catch (error) {
    console.error('Error deleting policy:', error);
    res.status(500).json({ error: 'Failed to delete policy' });
  }
});

// GET /api/retention-policies/:id/audit - Get policy audit log
router.get('/:id/audit', async (req, res) => {
  try {
    const { id } = req.params;
    const audit = await retentionPolicyService.getPolicyAudit(id);
    res.json({ audit });
  } catch (error) {
    console.error('Error fetching audit:', error);
    res.status(500).json({ error: 'Failed to fetch audit log' });
  }
});

export default router;
```
