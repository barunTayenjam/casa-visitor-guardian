```sql
-- Visitors table for management
CREATE TABLE IF NOT EXISTS visitors (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    type TEXT NOT NULL DEFAULT 'known', -- 'known' or 'unknown'

    -- Photo storage
    photo_path TEXT, -- Primary photo
    thumbnail_path TEXT, -- Thumbnail for UI

    -- Recognition data
    embedding_count INTEGER DEFAULT 0, -- Number of stored embeddings

    -- Statistics
    visit_count INTEGER DEFAULT 1,
    first_seen TIMESTAMP WITH TIME ZONE,
    last_seen TIMESTAMP WITH TIME ZONE,
    cameras_seen TEXT[], -- Array of camera IDs

    -- User notes
    notes TEXT,
    tags TEXT[], -- e.g., ['family', 'delivery', 'service']

    -- Status
    is_active BOOLEAN DEFAULT true,

    -- Timestamps
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Indexes
CREATE INDEX idx_visitors_name ON visitors(name);
CREATE INDEX idx_visitors_type ON visitors(type);
CREATE INDEX idx_visitors_active ON visitors(is_active);
CREATE INDEX idx_visitors_last_seen ON visitors(last_seen DESC);

-- Visitor events mapping table
CREATE TABLE IF NOT EXISTS visitor_events (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    visitor_id UUID NOT NULL REFERENCES visitors(id) ON DELETE CASCADE,
    event_id UUID NOT NULL REFERENCES events(id) ON DELETE CASCADE,
    face_id TEXT, -- Face detection ID in event
    confidence FLOAT, -- Recognition confidence
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(visitor_id, event_id)
);

CREATE INDEX idx_visitor_events_visitor ON visitor_events(visitor_id);
CREATE INDEX idx_visitor_events_event ON visitor_events(event_id);

-- Comments
COMMENT ON TABLE visitors IS 'Managed visitors with photos and metadata';
COMMENT ON TABLE visitor_events IS 'Mapping between visitors and events';
COMMENT ON COLUMN visitors.type IS 'Visitor type: known or unknown';
COMMENT ON COLUMN visitors.embedding_count IS 'Number of face embeddings stored for this visitor';
COMMENT ON COLUMN visitors.cameras_seen IS 'Array of camera IDs where visitor was detected';
```

```typescript
const router = express.Router();
const visitorRepository = AppDataSource.getRepository(Visitor);

// GET /api/visitors - List all visitors
router.get('/', async (req, res) => {
  try {
    const { type, search, page = '1', limit = '20' } = req.query;

    const queryBuilder = visitorRepository.createQueryBuilder('v')
      .where('v.isActive = :isActive', { isActive: true });

    if (type) {
      queryBuilder.andWhere('v.type = :type', { type });
    }

    if (search) {
      queryBuilder.andWhere(
        '(v.name ILIKE :search OR v.notes ILIKE :search)',
        { search: `%${search}%` }
      );
    }

    const page_num = parseInt(page as string);
    const limit_num = parseInt(limit as string);

    queryBuilder
      .orderBy('v.lastSeen', 'DESC')
      .skip((page_num - 1) * limit_num)
      .take(limit_num);

    const [visitors, total] = await queryBuilder.getManyAndCount();

    res.json({
      visitors,
      pagination: {
        page: page_num,
        limit: limit_num,
        total,
        pages: Math.ceil(total / limit_num)
      }
    });
  } catch (error) {
    console.error('Error fetching visitors:', error);
    res.status(500).json({ error: 'Failed to fetch visitors' });
  }
});

// GET /api/visitors/:id - Get visitor details
router.get('/:id', async (req, res) => {
  try {
    const { id } = req.params;

    const visitor = await visitorRepository.findOne({ where: { id } });
    if (!visitor) {
      return res.status(404).json({ error: 'Visitor not found' });
    }

    // Get recent events for this visitor
    const recentEvents = await visitorEventRepository.createQueryBuilder('ve')
      .innerJoinAndSelect('ve.event', 'event')
      .where('ve.visitorId = :visitorId', { visitorId: id })
      .orderBy('ve.createdAt', 'DESC')
      .limit(10)
      .getMany();

    res.json({
      ...visitor,
      recentEvents: recentEvents.map(ve => ({
        eventId: ve.eventId,
        timestamp: ve.event.timestamp,
        cameraId: ve.event.cameraId,
        confidence: ve.confidence,
        imagePath: ve.event.imagePath
      }))
    });
  } catch (error) {
    console.error('Error fetching visitor:', error);
    res.status(500).json({ error: 'Failed to fetch visitor' });
  }
});

// POST /api/visitors - Add new visitor from event
router.post('/', async (req, res) => {
  try {
    const { eventId, faceId, name, notes, tags } = req.body;

    if (!eventId || !faceId || !name) {
      return res.status(400).json({ error: 'eventId, faceId, and name are required' });
    }

    // Get event details
    const event = await eventRepository.findOne({ where: { id: eventId } });
    if (!event) {
      return res.status(404).json({ error: 'Event not found' });
    }

    // Extract face from event
    const faceDetections = event.faceDetections || [];
    const face = faceDetections.find((f: any) => f.id === faceId);

    if (!face) {
      return res.status(404).json({ error: 'Face detection not found in event' });
    }

    // Create visitor
    const visitor = visitorRepository.create({
      name,
      type: 'known',
      firstSeen: event.timestamp,
      lastSeen: event.timestamp,
      visitCount: 1,
      camerasSeen: [event.cameraId],
      notes: notes || null,
      tags: tags || [],
      embeddingCount: 0
    });

    await visitorRepository.save(visitor);

    // Map visitor to event
    const visitorEvent = visitorEventRepository.create({
      visitorId: visitor.id,
      eventId,
      faceId,
      confidence: face.confidence || 0
    });

    await visitorEventRepository.save(visitorEvent);

    // Extract and save face photo
    const photoPath = await _extractFacePhoto(event.imagePath, face.bbox, visitor.id);

    if (photoPath) {
      visitor.photoPath = photoPath;
      visitor.thumbnailPath = photoPath.replace('.jpg', '_thumb.jpg');
      await visitorRepository.save(visitor);
    }

    // Trigger embedding extraction from OpenCV service
    await _extractEmbedding(event.imagePath, face.bbox, visitor.id);

    res.status(201).json({
      id: visitor.id,
      name: visitor.name,
      photoPath: visitor.photoPath,
      message: 'Visitor added successfully'
    });
  } catch (error) {
    console.error('Error adding visitor:', error);
    res.status(500).json({ error: 'Failed to add visitor' });
  }
});

// PUT /api/visitors/:id - Update visitor
router.put('/:id', async (req, res) => {
  try {
    const { id } = req.params;
    const { name, notes, tags } = req.body;

    const visitor = await visitorRepository.findOne({ where: { id } });

// DELETE /api/visitors/:id - Delete visitor
router.delete('/:id', async (req, res) => {
  try {
    const { id } = req.params;

    const visitor = await visitorRepository.findOne({ where: { id } });
    if (!visitor) {
      return res.status(404).json({ error: 'Visitor not found' });
    }

    // Soft delete
    visitor.isActive = false;
    await visitorRepository.save(visitor);

    // Delete embeddings
    await AppDataSource.query(
      `UPDATE face_embeddings SET is_active = false WHERE visitor_id = $1`,

    res.json({ message: 'Visitor deleted successfully' });

// GET /api/visitors/:id/stats - Get visitor statistics
router.get('/:id/stats', async (req, res) => {

      .addSelect('MIN(ve.createdAt)', 'firstVisit')

      .groupBy('event.cameraId')
      .getRawMany();

      ...stats,
      cameraDistribution: cameraStats
    });
  } catch (error) {
    console.error('Error fetching stats:', error);
    res.status(500).json({ error: 'Failed to fetch statistics' });
  }
});
```
